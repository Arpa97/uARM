\newcommand{\fakethirtytwobitsscb}[1]{%
	\tiny
	\ifnum#1=1234567890
		#1
	\else
		\ifnum#1<3
			#1%
		\else
			\ifnum#1<8
				\ifnum#1=5
					$\cdots$%
				\fi
			\else
				\ifnum#1<13
					\count15=#1
					\advance\count15 by 5
					\the\count15%
				\else
					\ifnum#1=15
						$\cdots$%
					\fi
					\ifnum#1>17
						\count32=#1
						\advance\count32 by 12
						\the\count32%
					\fi
				\fi
			\fi
		\fi
	\fi
}
